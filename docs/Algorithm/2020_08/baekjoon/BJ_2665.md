## BOJ 2665 미로 만들기

[문제링크](https://www.acmicpc.net/problem/2665)

> [!TIP]
> 얼마전 풀었던 미로탈출을 풀어 그와 비슷하게 <b>bfs + 3차원 방문 체크</b>로 풀었더니 시간이 196ms? 메모리도 조금 컸음
>
> 그래서, <b>다익스트라 알고리즘</b>을 이용하여 풀었더니 무난하게 잘 풀렸다!
> 여기서 주의할 점은 <b>PriorityQueue</b>를 이용해야한다는 점!

```java
package baekjoon;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.PriorityQueue;

/**
 * @Author leejinyoung
 * @Date  Aug 30, 2020
 * @Language java
 * @Memory 13260KB
 * @RunningSpeed 88ms
 * @Description 다익스트라
 *
 */
public class BJ_2665_미로만들기 {

	private static char[][] map;
	private static int N;
	private static int[][] dir= {{1,0},{-1,0},{0,-1},{0,1}};
	private static int[][] wall;


	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(bf.readLine());

		map = new char[N][N];

		for (int i = 0; i < N; i++) {

			map[i] = bf.readLine().toCharArray();
		}

		bfs();

		System.out.println(wall[N-1][N-1]);

	}

	static class Point implements Comparable<Point>{
		int r,c,cnt;

		public Point(int r, int c, int cnt) {
			super();
			this.r = r;
			this.c = c;
			this.cnt = cnt;
		}

		@Override
		public int compareTo(Point o) {

			return Integer.compare(this.cnt, o.cnt);
		}

	}

	private static void bfs() {
		wall = new int[N][N];

		for (int i = 0; i < N; i++) {
			Arrays.fill(wall[i], -1);
		}
		PriorityQueue<Point> q = new PriorityQueue<>();
		q.offer(new Point(0,0,0));
		wall[0][0]=0;


		while(!q.isEmpty()) {
			Point f = q.poll();

			if(f.r==N-1 && f.c==N-1) {
				break;
			}

			for (int i = 0; i < 4; i++) {
				int nr = f.r +dir[i][0];
				int nc = f.c +dir[i][1];

				if(nr<0 || nr>=N || nc<0 || nc>=N) continue;
				if(map[nr][nc] == '0') {
					if(wall[nr][nc]!=-1 && wall[nr][nc] <= f.cnt+1) continue;
					q.offer(new Point(nr,nc,f.cnt+1));
					wall[nr][nc] = f.cnt+1;

				}else {
					if(wall[nr][nc]!=-1 && wall[nr][nc] <= f.cnt) continue;
					q.offer(new Point(nr,nc,f.cnt));
					wall[nr][nc] = f.cnt;
				}


			}




		}


	}

}
```
